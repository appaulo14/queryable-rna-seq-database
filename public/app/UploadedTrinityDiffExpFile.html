def get_next_line
  line = @uploaded_file.tempfile.readline.strip
  return nil if line.blank?
  (item, logConc, log_fold_change, p_value, fdr) = line.split(/\s+/)
  return TrinityDiffExpLine.new(item,log_fold_change,p_value, fdr)
end
